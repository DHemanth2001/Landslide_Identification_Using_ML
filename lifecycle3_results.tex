% ---- Slide 2: Feature Extraction — Handcrafted vs Learned ----
\begin{frame}{Why Deep Learning Over Handcrafted Features}
    \textbf{Handcrafted approach:} Manually design (NDVI, GLCM, edges, etc.) $\to$ feed to SVM/Random Forest
    \begin{itemize}
        \item Requires domain expert knowledge
        \item Cannot capture patterns the designer didn't anticipate
        \item Baseline: SVM 84\%, Random Forest 88\%, CNN 92.5\% (Wang et al., 2021)
    \end{itemize}

    \vspace{0.2cm}
    \textbf{Deep learning approach:} CNN/Transformer learns features automatically
    \begin{itemize}
        \item Edges $\to$ textures $\to$ patterns $\to$ landslide semantics (layer by layer)
        \item \textbf{Our result:} 98.92\% on Bijie (21x advantage on single-source)
    \end{itemize}
\end{frame}

% ---- Slide 3: What We Did in LC1 and LC2 ----
\begin{frame}{Recap: Lifecycle 1 and 2}
    \textbf{Lifecycle 1 --- Foundation (HR-GLDD, 3,439 images):}
    \begin{itemize}
        \item r0: AlexNet from scratch --- 78.54\% accuracy, 85.53\% AUC
        \item r1: AlexNet + ImageNet transfer learning --- 80.11\%, 85.73\% AUC
        \item r2: Replaced with EfficientNet-B3 --- 79.97\%, \textbf{recall 67\% $\to$ 78\%}, AUC 88.93\%

    \vspace{0.3cm}
    \textbf{Lifecycle 2 --- Optimization (same HR-GLDD dataset):}
    \begin{itemize}
        \item r3: Added temperature scaling for calibration --- fixed overconfidence
        \item r4: EfficientNet + AlexNet ensemble (CNN+CNN) --- 82.40\%, AUC 89.83\%
        \item r5: EfficientNet + ViT ensemble (CNN+Transformer) --- \textbf{84.55\%}, AUC 91.50\%
        \item Key learning: CNN+Transformer beats CNN+CNN because \textbf{different architectures catch different errors}
    \end{itemize}

    \vspace{0.3cm}
    \textbf{Gap after LC2:} 84.55\% is still below published results (85--92\%). Limited by small single-source dataset and older architectures.
\end{frame}

% ---- Slide 7: Model 1 --- EfficientNetV2 + CBAM ----
\begin{frame}{LC3 Model 1: EfficientNetV2-S + CBAM}
    \textbf{What it is:} Upgraded EfficientNet-B3 from LC1. Input: 384×384.

    \vspace{0.2cm}
    \textbf{Key improvements:}
    \begin{itemize}
        \item \textbf{Fused MBConv blocks:} Combines expansion + depthwise conv (faster than B3)
        \item \textbf{CBAM attention:} Channel attention (learns which features matter) + Spatial attention (learns where to look)
    \end{itemize}

    \vspace{0.2cm}
    \textbf{Accuracy:} 84.50\% (multi-source), 98.20\% (Bijie)
\end{frame}

% ---- Slide 8: Model 2 --- ConvNeXt + CBAM + FPN ----
\begin{frame}{LC3 Model 2: ConvNeXt-Base + CBAM + FPN (Best Model)}
    \textbf{What it is:} Most powerful pure CNN (Liu et al., CVPR 2022). Input: 224$\times$224.

    \vspace{0.2cm}
    \textbf{Key innovation:}
    \begin{itemize}
        \item \textbf{ConvNeXt backbone:} 4 stages (128 → 256 → 512 → 1024 channels)
        \item \textbf{FPN (Feature Pyramid Network):} Multi-scale fusion from all 4 stages
        \item \textbf{CBAM at each scale:} Attention per resolution level
        \item \textbf{Why best:} Multi-scale + attention at every level = detects landslides of \textbf{any size}
    \end{itemize}

    \vspace{0.2cm}
    \textbf{Accuracy:} 87.85\% (multi-source), 98.56\% (Bijie)
\end{frame}

% ---- Slide 9: Model 3 --- SwinV2 ----
\begin{frame}{LC3 Model 3: SwinV2-Small (Vision Transformer)}
    \textbf{What it is:} Hierarchical Vision Transformer (Liu et al., CVPR 2022). Input: 256$\times$256.

    \vspace{0.2cm}
    \textbf{Key differences from CNN:}
    \begin{itemize}
        \item \textbf{Patch embedding:} Splits image into 4×4 patches (each patch = one token)
        \item \textbf{Self-attention:} Every patch looks at every other patch (global context)
        \item \textbf{Hierarchical stages:} 4 stages with patch merging between them
        \item \textbf{Why it works:} CNNs see only 3×3 neighborhoods; Transformers see entire windows
    \end{itemize}

    \vspace{0.2cm}
    \textbf{Example:} Bare soil near river = non-landslide; bare soil on steep slope = landslide. Requires global context to distinguish.

    \vspace{0.2cm}
    \textbf{Accuracy:} 88.03\% (multi-source), 97.48\% (Bijie)
\end{frame}

% ---- Slide 15: Phase 2 Introduction ----
\begin{frame}{Phase 2: Temporal Forecasting using Hidden Markov Models}
    \textbf{After Phase 1 detects a landslide, Phase 2 asks:}
    \begin{itemize}
        \item How \textbf{likely} is another occurrence? (probability with dotted line tracking)
        \item When is the \textbf{peak risk month}? (seasonal pattern)
        \item What types will occur \textbf{over the next 3 months}? (forecast)
    \end{itemize}

    \vspace{0.4cm}
    \textbf{Data Source:} NASA Global Landslide Catalog --- \textbf{Filtered to Bijie, China}
    \begin{itemize}
        \item 13+ historical landslide events in Bijie region
        \item 28 years of temporal history (1987--2015)
        \item Hidden states model geological regimes; observations encode occurrence sequences
    \end{itemize}

    \vspace{0.3cm}
    \textbf{Key Innovation:} Combines \textbf{spatial detection} (Phase 1 CNN) with \textbf{temporal forecasting} (Phase 2 HMM)
\end{frame}

% ---- Slide 16: HMM Architecture ----
\begin{frame}{HMM Architecture: How It Works}
    \textbf{Hidden Markov Model = Sequence Prediction with Interpretability}
    
    \begin{itemize}
        \item \textbf{Hidden States (8):} Latent regimes (e.g., high-risk monsoon season, dormant period)
        \item \textbf{Observations:} Landslide occurrences in Bijie sequence
        \item \textbf{Transition Matrix:} Probability of moving state-to-state (``monsoon $\to$ dormant'')
        \item \textbf{Training:} Baum-Welch algorithm on 13 historical events
        \item \textbf{Inference:} Viterbi decoding to find most likely sequence \& forecast next 3 months
    \end{itemize}

    \vspace{0.4cm}
    \textbf{Why HMM?} Simple, fast, fully interpretable. Outputs probabilities you can understand and explain to stakeholders.
\end{frame}

% ---- Slide 18: HMM Prediction Pipeline ----
\begin{frame}{Phase 2 Prediction Pipeline: Simple Flow}
    \textbf{Given a detected landslide in Bijie:}

    \vspace{0.3cm}
    \begin{enumerate}
        \item \textbf{Extract location} from Phase 1 detection
        \item \textbf{Query Bijie history:} Retrieve 13+ similar events from NASA GLC
        \item \textbf{Run HMM inference:} Viterbi decoder finds most likely state sequence
        \item \textbf{Compute probabilities:}
            \begin{itemize}
                \item Occurrence probability in next 6 months
                \item Peak risk month (seasonal multiplier)
                \item 3-month forecast with dotted-line timeline
            \end{itemize}
        \item \textbf{Generate report:} Text summary + SVG hotspot map
    \end{enumerate}

    \vspace{0.3cm}
    \textbf{Execution:} $<2$ seconds per query. Production-ready Python code.
\end{frame}
